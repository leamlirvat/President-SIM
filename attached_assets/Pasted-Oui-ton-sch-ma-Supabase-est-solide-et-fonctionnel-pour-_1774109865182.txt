export async function calculate_economy(event: any) {
  const { data: countries } = await supabase
    .from('countries')
    .select('*')
    .eq('serverid', event.serverid);
    
  for (const country of countries) {
    // Calculs complexes: bâtiments, impôts, corruption, dette
    const income = calculate_income(country.tiles, country.corruption);
    const expenses = calculate_expenses(country.army, country.population);
    
    await supabase
      .from('countries')
      .update({ 
        money: Math.max(0, country.money + income - expenses),

      })

}

sql
-- 1. CORE (existant à compléter)
ALTER TABLE countries 
ADD COLUMN population BIGINT DEFAULT 1000000,
ADD COLUMN level INTEGER DEFAULT 1,
ADD COLUMN prestige INTEGER DEFAULT 0,
ADD COLUMN debt NUMERIC DEFAULT 0,
ADD COLUMN popularity NUMERIC DEFAULT 75 CHECK (popularity >= 0 AND popularity <= 100),
ADD COLUMN corruption NUMERIC DEFAULT 20 CHECK (corruption >= 0 AND corruption <= 100),
ADD COLUMN last_active TIMESTAMP DEFAULT NOW();

-- 2. TERRITOIRE (amélioré)
ALTER TABLE tiles 
ADD COLUMN resource_bonus JSONB DEFAULT '{}',
ADD COLUMN buildings JSONB DEFAULT '[]'::JSONB,
ADD COLUMN stability INTEGER DEFAULT 100 CHECK (stability >= 0 AND stability <= 100),
ADD COLUMN discovered_by INTEGER[] DEFAULT '{}';

CREATE TABLE buildings (
  id BIGSERIAL PRIMARY KEY,
  tile_id BIGINT REFERENCES tiles(id),
  country_id BIGINT REFERENCES countries(id),
  type TEXT NOT NULL, -- usine_fer, centrale_nucleaire, base_militaire
  level INTEGER DEFAULT 1 CHECK (level >= 1 AND level <= 5),
  production_rate NUMERIC DEFAULT 1,
  energy_cost NUMERIC DEFAULT 0,
  workers_needed INTEGER DEFAULT 100,
  last_maintenance TIMESTAMP DEFAULT NOW()
);

-- 3. RESSOURCES (système complet)
CREATE TABLE resources (
  id BIGSERIAL PRIMARY KEY,
  country_id BIGINT REFERENCES countries(id) ON DELETE CASCADE,
  resource_type TEXT NOT NULL, -- fer, petrole, uranium, puces, etc.
  quantity NUMERIC DEFAULT 0,
  production_per_hour NUMERIC DEFAULT 0,
  consumption_per_hour NUMERIC DEFAULT 0,
  storage_capacity NUMERIC DEFAULT 10000,
  market_price NUMERIC DEFAULT 1,
  UNIQUE(country_id, resource_type)
);

-- 4. ALLIANCES (système modulaire)
CREATE TABLE alliances (
  id BIGSERIAL PRIMARY KEY,
  serverid INTEGER NOT NULL,
  name TEXT NOT NULL,
  color TEXT NOT NULL,
  leader_country_id BIGINT REFERENCES countries(id),
  member_limit INTEGER DEFAULT 10 CHECK (member_limit >= 3 AND member_limit <= 50),
  member_count INTEGER DEFAULT 1,
  creation_date TIMESTAMP DEFAULT NOW(),
  activity_level NUMERIC DEFAULT 100 CHECK (activity_level >= 0 AND activity_level <= 100)
);

CREATE TABLE alliance_members (
  id BIGSERIAL PRIMARY KEY,
  alliance_id BIGINT REFERENCES alliances(id) ON DELETE CASCADE,
  country_id BIGINT REFERENCES countries(id) ON DELETE CASCADE,
  role TEXT DEFAULT 'membre' CHECK (role IN ('leader', 'co-leader', 'ministre_eco', 'ministre_def', 'membre')),
  join_date TIMESTAMP DEFAULT NOW(),
  permissions JSONB DEFAULT '{}',
  UNIQUE(alliance_id, country_id)
);

CREATE TABLE alliance_modules (
  id BIGSERIAL PRIMARY KEY,
  alliance_id BIGINT REFERENCES alliances(id) ON DELETE CASCADE,
  module_code TEXT NOT NULL, -- DEFENSE_PACT, COMMON_MARKET, RESEARCH_POOL
  level INTEGER DEFAULT 1 CHECK (level >= 1 AND level <= 5),
  activation_date TIMESTAMP DEFAULT NOW(),
  duration_days INTEGER DEFAULT 30
);

-- 5. RLS SÉCURITÉ (CRITIQUE)
ALTER TABLE countries ENABLE ROW LEVEL SECURITY;
ALTER TABLE tiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE buildings ENABLE ROW LEVEL SECURITY;
ALTER TABLE resources ENABLE ROW LEVEL SECURITY;
ALTER TABLE alliances ENABLE ROW LEVEL SECURITY;
ALTER TABLE alliance_members ENABLE ROW LEVEL SECURITY;

-- Policies RLS
CREATE POLICY "Users see own country" ON countries FOR SELECT USING (auth.uid() = userid);
CREATE POLICY "Users update own country" ON countries FOR UPDATE USING (auth.uid() = userid);
CREATE POLICY "Users insert own country" ON countries FOR INSERT WITH CHECK (auth.uid() = userid);

CREATE POLICY "Users see own server tiles" ON tiles FOR ALL 
USING ((SELECT serverid FROM countries WHERE userid = auth.uid()) = serverid);

CREATE POLICY "Alliance members see alliance" ON alliances FOR SELECT 
USING (id IN (SELECT alliance_id FROM alliance_members WHERE country_id IN 
  (SELECT id FROM countries WHERE userid = auth.uid())));

-- 6. Index performance
CREATE INDEX idx_countries_server_user ON countries(serverid, userid);
CREATE INDEX idx_tiles_server_owner ON tiles(serverid, ownercountryid);
CREATE INDEX idx_alliances_server ON alliances(serverid);
